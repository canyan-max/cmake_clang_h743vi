~~~
typedef struct ST7789_SPI_OPS_T
{
    st7789_state_t (*pf_spi_transmit)(uint8_t *data,                    
                                     uint32_t          lenth);
    st7789_state_t (*pf_backlight_pin)(uint8_t          on);
    st7789_state_t (*pf_dataorcmd_pin)(uint8_t          on);
    void          (*pf_delay_ms)(uint32_t ms);                                  // Delay milliseconds.
} st7789_spi_ops_t;

static st7789_state_t spi_transmit( uint8_t         *p_data,
                                    uint32_t         data_len)
{
    uint32_t hal_ret = HAL_StatusTypeDef hal_ret = HAL_SPI_Transmit(
        &hspi4,
        p_data, data_len, 100U);
    if(HAL_OK!= hal_ret)
    {
        return ST7789_ERROR;
    }
    return ST7789_OK;
}


static st7789_state_t dc_pin_en( uint8_t         on)
{
    GPIO_PinState pin_state = (0U != on) ? GPIO_PIN_SET : GPIO_PIN_RESET;
    HAL_GPIO_WritePin(SPI4_DC_GPIO_Port,
                      SPI4_DC_Pin, GPIO_PIN_SET);
    return ST7789_OK;
}
